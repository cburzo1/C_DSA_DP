struct TNode{
    int data;
    struct TNode *lch;
    struct TNode *rch;
}*root = NULL;

struct QNode{
    struct TNode *data;
    struct QNode *next;
}*head =NULL, *tail = NULL;

void enQ(struct TNode *val){
    struct QNode *newNode = (struct QNode*)malloc(sizeof(struct QNode));
    
    newNode->data = val;
    
    if(head == NULL){
        head = newNode;
        tail = newNode;
    }else{
        tail->next = newNode;
        tail = newNode;
    }
}
